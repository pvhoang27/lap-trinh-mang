#include<bits/stdc++.h>
	using namespace std;
	int main(){
		int a[]={1,3,4,19,33,20,55,34};
		int Min=99999;
		sort(a,a+8);
		for(int i=0;i<8;i++){
			for(int j=i+1;j<8;j++){
				Min=min(abs(a[j]-a[i]),Min);
			}
		}
		int index1,index2;
		for(int i=0;i<8;i++){
			for(int j=i+1;j<8;j++){
				if(abs(a[j]-a[i])==Min){
					index1=a[i];
					index2=a[j];
				}
			}
		}
		cout<<Min<<" "<<index1<<" "<<index2;
	}
